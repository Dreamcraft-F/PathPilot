# PathPilot 无感历史捕获引擎实现方案

## 1. 技术选型

### 1.1 开发语言和框架
- **主语言**：C# (.NET 6/7) 或 Python
- **框架选择**：
  - **C#方案**：WPF（界面）+ Win32 API（系统交互）
  - **Python方案**：PyQt/Tkinter（界面）+ pywin32（系统交互）
- **推荐**：C#方案，性能更好，与Windows集成更紧密

### 1.2 系统交互技术
- **窗口监听**：Windows UI Automation API
- **路径获取**：Shell32 API + UI Automation
- **事件钩子**：SetWinEventHook（窗口事件监听）
- **进程管理**：Windows Management Instrumentation (WMI)

### 1.3 数据存储
- **主数据库**：SQLite（轻量级，无需安装）
- **缓存**：内存缓存（Dictionary或ConcurrentDictionary）
- **配置文件**：JSON格式

string GetExplorerPath(IntPtr hwnd)
{
    try
    {
        AutomationElement element = AutomationElement.FromHandle(hwnd);
        
        // 查找地址栏
        var addressBar = element.FindFirst(TreeScope.Descendants,
            new PropertyCondition(AutomationElement.AutomationIdProperty, "1001"));
        
        if (addressBar != null)
        {
            // 获取地址栏的值
            var valuePattern = addressBar.GetCurrentPattern(ValuePattern.Pattern) 
                as ValuePattern;
            return valuePattern.Current.Value;
        }
    }
    catch (Exception ex)
    {
        // 处理异常
        LogError(ex);
    }
    return null;
}

### 2.3 路径处理流水线

#### 2.3.1 处理流水线架构
```csharp
class PathProcessingPipeline
{
    private List<IPathFilter> filters;
    private IDeduplicator deduplicator;
    private IImportanceCalculator importanceCalculator;
    
    public PathRecord ProcessPath(string rawPath, IntPtr hwnd)
    {
        // 1. 路径规范化
        string normalizedPath = NormalizePath(rawPath);
        
        // 2. 应用过滤规则
        if (!ShouldRecordPath(normalizedPath))
            return null;
        
        // 3. 去重处理
        if (deduplicator.IsDuplicate(normalizedPath))
            return null;
        
        // 4. 计算重要性
        int importance = importanceCalculator.Calculate(normalizedPath);
        
        // 5. 创建记录
        return new PathRecord
        {
            Path = normalizedPath,
            FirstVisit = DateTime.Now,
            LastVisit = DateTime.Now,
            ImportanceScore = importance
        };
    }
    
    private string NormalizePath(string path)
    {
        // 转换为小写
        path = path.ToLower();
        
        // 统一路径分隔符
        path = path.Replace('/', '\\');
        
        // 去除尾部斜杠
        if (path.EndsWith("\\"))
            path = path.TrimEnd('\\');
        
        // 解析环境变量
        path = Environment.ExpandEnvironmentVariables(path);
        
        return path;
    }
}
```

### 2.4 智能去重算法

#### 2.4.1 基于时间的去重
```csharp
class TimeBasedDeduplicator : IDeduplicator
{
    private readonly ConcurrentDictionary<string, DateTime> lastAccessTimes;
    private readonly TimeSpan timeWindow = TimeSpan.FromSeconds(5);
    
    public bool IsDuplicate(string path)
    {
        DateTime now = DateTime.Now;
        
        if (lastAccessTimes.TryGetValue(path, out DateTime lastAccess))
        {
            // 在时间窗口内，视为重复
            if (now - lastAccess < timeWindow)
                return true;
        }
        
        // 更新最后访问时间
        lastAccessTimes[path] = now;
        return false;
    }
}
```

#### 2.4.2 基于层级的智能过滤
```csharp
class HierarchicalDeduplicator : IDeduplicator
{
    private readonly List<string> recentPaths = new List<string>();
    private readonly int maxRecentPaths = 10;
    
    public bool IsDuplicate(string path)
    {
        if (recentPaths.Count == 0)
        {
            recentPaths.Add(path);
            return false;
        }
        
        string lastPath = recentPaths.Last();
        
        // 检查是否是向上导航
        if (IsParentPath(lastPath, path))
        {
            // 从子目录返回到父目录
            // 检查是否已经记录了父目录
            if (recentPaths.Contains(path))
            {
                // 更新顺序，移到最新
                recentPaths.Remove(path);
                recentPaths.Add(path);
                return true; // 已经记录过
            }
        }
        // 检查是否是向下导航
        else if (IsParentPath(path, lastPath))
        {
            // 从父目录进入子目录
            // 检查是否已经记录了子目录
            if (recentPaths.Contains(path))
            {
                recentPaths.Remove(path);
                recentPaths.Add(path);
                return true;
            }
        }
        
        // 新路径，添加到列表
        recentPaths.Add(path);
        
        // 保持列表大小
        if (recentPaths.Count > maxRecentPaths)
            recentPaths.RemoveAt(0);
        
        return false;
    }
    
    private bool IsParentPath(string parent, string child)
    {
        // 检查child是否是parent的子路径
        return child.StartsWith(parent + "\\") || child.StartsWith(parent);
    }
}
```

### 2.5 重要性评分算法

```csharp
class ImportanceCalculator
{
    public int Calculate(PathRecord record)
    {
        int score = 100; // 基础分
        
        // 时间权重：停留时间越长，分数越高
        double timeWeight = Math.Min(record.TotalTimeSpent / 60000.0, 1.0); // 最大1分钟
        score += (int)(timeWeight * 50);
        
        // 频率权重：访问次数越多，分数越高
        double freqWeight = Math.Min(record.VisitCount / 10.0, 1.0); // 最大10次
        score += (int)(freqWeight * 30);
        
        // 深度权重：路径深度适中（3-5层）加分
        int depth = record.Path.Count(c => c == '\\');
        if (depth >= 3 && depth <= 5)
            score += 20;
        else if (depth > 5)
            score -= 10;
        
        // 收藏加分
        if (record.IsFavorite)
            score += 100;
        
        return score;
    }
}
```

class AsyncPathProcessor
{
    private readonly BlockingCollection<PathEvent> eventQueue;
    private readonly CancellationTokenSource cts;
    private readonly Task processingTask;
    
    public AsyncPathProcessor()
    {
        eventQueue = new BlockingCollection<PathEvent>();
        cts = new CancellationTokenSource();
        processingTask = Task.Factory.StartNew(ProcessEvents, 
            cts.Token, TaskCreationOptions.LongRunning, TaskScheduler.Default);
    }
    
    public void EnqueuePathEvent(PathEvent pathEvent)
    {
        eventQueue.Add(pathEvent);
    }
    
    private void ProcessEvents()
    {
        foreach (var pathEvent in eventQueue.GetConsumingEnumerable(cts.Token))
        {
            // 在后台线程处理路径事件
            ProcessPathAsync(pathEvent);
        }
    }
}

### 4.2 缓存策略
```csharp
class PathCache
{
    private readonly ConcurrentDictionary<string, PathRecord> cache;
    private readonly int maxCacheSize = 1000;
    
    public PathRecord GetOrAdd(string path, Func<PathRecord> factory)
    {
        return cache.GetOrAdd(path, key =>
        {
            var record = factory();
            
            // 检查缓存大小
            if (cache.Count >= maxCacheSize)
            {
                // 清理最旧的记录
                CleanupOldEntries();
            }
            
            return record;
        });
    }
    
    private void CleanupOldEntries()
    {
        // 按最后访问时间排序，移除最旧的20%记录
        var entriesToRemove = cache
            .OrderBy(kvp => kvp.Value.LastVisit)
            .Take(maxCacheSize / 5)
            .Select(kvp => kvp.Key);
        
        foreach (var key in entriesToRemove)
        {
            cache.TryRemove(key, out _);
        }
    }
}
```

### 4.3 批量写入
```csharp
class BatchWriter
{
    private readonly List<PathRecord> pendingRecords = new List<PathRecord>();
    private readonly Timer flushTimer;
    private readonly int batchSize = 50;
    private readonly TimeSpan flushInterval = TimeSpan.FromSeconds(10);
    
    public BatchWriter()
    {
        flushTimer = new Timer(FlushPendingRecords, null, 
            flushInterval, flushInterval);
    }
    
    public void AddRecord(PathRecord record)
    {
        lock (pendingRecords)
        {
            pendingRecords.Add(record);
            
            if (pendingRecords.Count >= batchSize)
            {
                FlushPendingRecords(null);
            }
        }
    }
    
    private void FlushPendingRecords(object state)
    {
        List<PathRecord> recordsToWrite;
        
        lock (pendingRecords)
        {
            if (pendingRecords.Count == 0) return;
            
            recordsToWrite = new List<PathRecord>(pendingRecords);
            pendingRecords.Clear();
        }
        
        // 批量写入数据库
        using var transaction = database.BeginTransaction();
        foreach (var record in recordsToWrite)
        {
            database.InsertPathRecord(record);
        }
        transaction.Commit();
    }
}
```
