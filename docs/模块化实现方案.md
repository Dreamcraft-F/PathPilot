# PathPilot 模块化实现方案

## 1. 模块划分概述

PathPilot 采用模块化设计，将整个系统划分为以下核心模块：

### 1.1 核心模块
1. **应用主模块** (app.py) - 应用生命周期管理
2. **无感历史捕获引擎** (path_engine.py) - 核心业务逻辑
3. **数据库管理模块** (database/) - 数据存储和管理
4. **GUI模块** (gui/) - 用户界面交互
5. **配置管理模块** (config/) - 配置文件处理
6. **工具模块** (utils/) - 通用工具函数

### 1.2 模块依赖关系
```
应用主模块 (app.py)
├── 无感历史捕获引擎 (path_engine.py)
│   ├── 窗口监控器 (window_monitor.py)
│   ├── 路径过滤器 (path_filter.py)
│   ├── 去重处理器 (path_dedup.py)
│   └── 重要性计算器 (importance_calc.py)
├── 数据库管理模块 (database/)
│   ├── 数据库管理器 (db_manager.py)
│   └── 数据模型 (models.py)
├── GUI模块 (gui/)
│   ├── 主窗口 (main_window.py)
│   ├── 系统托盘图标 (tray_icon.py)
│   ├── 悬浮球 (floating_ball.py)
│   └── 搜索组件 (search_widget.py)
├── 配置管理模块 (config/)
│   └── 配置管理器 (config_manager.py)
└── 工具模块 (utils/)
    ├── Windows API工具 (win32_utils.py)
    ├── 路径处理工具 (path_utils.py)
    └── 日志工具 (logger.py)
```

## 2. 模块实现顺序

### 2.1 第一阶段：基础框架
1. **配置管理模块** - 首先实现，为其他模块提供配置支持
2. **工具模块** - 提供通用工具函数
3. **数据库管理模块** - 数据存储基础

### 2.2 第二阶段：核心引擎
1. **无感历史捕获引擎** - 核心业务逻辑
   - 窗口监控器
   - 路径过滤器
   - 去重处理器
   - 重要性计算器

### 2.3 第三阶段：用户界面
1. **GUI模块** - 用户交互界面
   - 系统托盘图标
   - 悬浮球
   - 主窗口
   - 搜索组件

### 2.4 第四阶段：集成测试
1. **应用主模块** - 整合所有模块
2. **集成测试** - 测试完整功能

```python
class ConfigManager:
    """配置管理器"""
    
    def __init__(self, config_dir="config"):
        self.config_dir = config_dir
        self.config_file = os.path.join(config_dir, "user_config.json")
        self.default_config_file = os.path.join(config_dir, "default_config.json")
        self.config = {}
        
    def load(self):
        """加载配置"""
        # 1. 加载默认配置
        default_config = self.load_json(self.default_config_file)
        
        # 2. 加载用户配置（如果存在）
        user_config = {}
        if os.path.exists(self.config_file):
            user_config = self.load_json(self.config_file)
            
        # 3. 合并配置（用户配置覆盖默认配置）
        self.config = self.merge_config(default_config, user_config)
        
    def get(self, key, default=None):
        """获取配置项"""
        keys = key.split('.')
        value = self.config
        for k in keys:
            if isinstance(value, dict) and k in value:
                value = value[k]
            else:
                return default
        return value
        
    def set(self, key, value):
        """设置配置项"""
        keys = key.split('.')
        config = self.config
        for k in keys[:-1]:
            if k not in config:
                config[k] = {}
            config = config[k]
        config[keys[-1]] = value
        
    def save(self):
        """保存配置"""
        self.save_json(self.config_file, self.config)
```

```python
import win32gui

import psutil

class Win32Utils:
    """Windows API工具类"""
    
    @staticmethod
    def get_window_class_name(hwnd):
        """获取窗口类名"""
        return win32gui.GetClassName(hwnd)
        
    @staticmethod
    def get_window_process_name(hwnd):
        """获取窗口进程名"""
        _, pid = win32process.GetWindowThreadProcessId(hwnd)
        try:
            process = psutil.Process(pid)
            return process.name()
        except:
            return None
            
    @staticmethod
    def is_explorer_window(hwnd):
        """判断是否是资源管理器窗口"""
        class_name = Win32Utils.get_window_class_name(hwnd)
        if class_name == "CabinetWClass":
            return True
            
        process_name = Win32Utils.get_window_process_name(hwnd)
        return process_name and process_name.lower() == "explorer.exe"
        
    @staticmethod
    def get_foreground_window():
        """获取前台窗口"""
        return win32gui.GetForegroundWindow()
```

```python
import os
import re

class PathUtils:
    """路径处理工具类"""
    
    @staticmethod
    def normalize_path(path):
        """规范化路径"""
        # 转换为小写
        path = path.lower()
        
        # 统一路径分隔符
        path = path.replace('/', '\\')
        
        # 去除尾部斜杠
        if path.endswith('\\'):
            path = path.rstrip('\\')
            
        # 解析环境变量
        path = os.path.expandvars(path)
        
        return path
        
    @staticmethod
    def get_path_depth(path):
        """获取路径深度"""
        return path.count('\\')
        
    @staticmethod
    def is_parent_path(parent, child):
        """判断是否是父路径"""
        parent = parent.lower()
        child = child.lower()
        return child.startswith(parent + '\\') or child == parent
        
    @staticmethod
    def matches_pattern(path, pattern):
        """匹配路径模式"""
        # 支持通配符 * 和 ?
        regex = pattern.replace('*', '.*').replace('?', '.')
        return re.match(regex, path, re.IGNORECASE) is not None
```

```python
import logging

class Logger:
    """日志工具类"""

        self.logger = None
        self.setup_logger()

        os.makedirs(self.log_dir, exist_ok=True)
        
        # 创建日志器

        self.logger.setLevel(logging.DEBUG)
```

#### 3.3.2 数据库管理器 (db_manager.py)
```python
import sqlite3
import os
from typing import List, Optional
from datetime import datetime, timedelta

class DatabaseManager:
    """数据库管理器"""
    
    def __init__(self, db_path="data/pathpilot.db"):
        self.db_path = db_path
        self.connection = None
        
    def initialize(self):
        """初始化数据库"""
        # 确保目录存在
        os.makedirs(os.path.dirname(self.db_path), exist_ok=True)
        
        # 连接数据库
        self.connection = sqlite3.connect(self.db_path)
        self.connection.row_factory = sqlite3.Row
        
        # 创建表
        self.create_tables()
        
    def create_tables(self):
        """创建数据库表"""
        cursor = self.connection.cursor()
        
        # 创建路径记录表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS path_records (
                id TEXT PRIMARY KEY,
                path TEXT NOT NULL,
                first_visit DATETIME NOT NULL,
                last_visit DATETIME NOT NULL,
                visit_count INTEGER DEFAULT 1,
                total_time_spent INTEGER DEFAULT 0,
                importance_score INTEGER DEFAULT 100,
                is_favorite BOOLEAN DEFAULT FALSE,
                tags TEXT DEFAULT '[]',
                source TEXT DEFAULT 'auto',
                status TEXT DEFAULT 'active',
                created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
                updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        # 创建访问事件表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS visit_events (
                id INTEGER PRIMARY KEY AUTOINCREMENT,
                path TEXT NOT NULL,
                timestamp DATETIME NOT NULL,
                duration INTEGER DEFAULT 0,
                window_id TEXT,
                previous_path TEXT,
                navigation_type TEXT DEFAULT 'direct',
                created_at DATETIME DEFAULT CURRENT_TIMESTAMP
            )
        """)
        
        # 创建索引
        cursor.execute("CREATE INDEX IF NOT EXISTS idx_path_records_path ON path_records(path)")
        cursor.execute("CREATE INDEX IF NOT EXISTS idx_path_records_last_visit ON path_records(last_visit)")
        cursor.execute("CREATE INDEX IF NOT EXISTS idx_path_records_importance ON path_records(importance_score)")
        cursor.execute("CREATE INDEX IF NOT EXISTS idx_visit_events_timestamp ON visit_events(timestamp)")
        
        self.connection.commit()
        
    def insert_path_record(self, record):
        """插入路径记录"""
        cursor = self.connection.cursor()
        cursor.execute("""
            INSERT INTO path_records 
            (id, path, first_visit, last_visit, visit_count, total_time_spent, 
             importance_score, is_favorite, tags, source, status)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
        """, (
            record.id, record.path, record.first_visit, record.last_visit,
            record.visit_count, record.total_time_spent, record.importance_score,
            record.is_favorite, str(record.tags), record.source, record.status
        ))
        self.connection.commit()
        
    def update_path_record(self, record):
        """更新路径记录"""
        cursor = self.connection.cursor()
        cursor.execute("""
            UPDATE path_records 
            SET last_visit = ?, visit_count = ?, total_time_spent = ?,
                importance_score = ?, is_favorite = ?, tags = ?, status = ?,
                updated_at = CURRENT_TIMESTAMP
            WHERE id = ?
        """, (
            record.last_visit, record.visit_count, record.total_time_spent,
            record.importance_score, record.is_favorite, str(record.tags), 
            record.status, record.id
        ))
        self.connection.commit()
        
    def get_path_record_by_path(self, path) -> Optional[PathRecord]:
        """根据路径获取记录"""
        cursor = self.connection.cursor()
        cursor.execute("SELECT * FROM path_records WHERE path = ?", (path,))
        row = cursor.fetchone()
        if row:
            return self.row_to_record(row)
        return None
        
    def get_recent_paths(self, limit=100) -> List[PathRecord]:
        """获取最近访问的路径"""
        cursor = self.connection.cursor()
        cursor.execute("""
            SELECT * FROM path_records 
            WHERE status = 'active'
            ORDER BY last_visit DESC 
            LIMIT ?
        """, (limit,))
        rows = cursor.fetchall()
        return [self.row_to_record(row) for row in rows]
        
    def search_paths(self, keyword, limit=50) -> List[PathRecord]:
        """搜索路径"""
        cursor = self.connection.cursor()
        cursor.execute("""
            SELECT * FROM path_records 
            WHERE path LIKE ? AND status = 'active'
            ORDER BY importance_score DESC, last_visit DESC
            LIMIT ?
        """, (f'%{keyword}%', limit))
        rows = cursor.fetchall()
        return [self.row_to_record(row) for row in rows]
        
    def delete_path_record(self, path_id):
        """删除路径记录（软删除）"""
        cursor = self.connection.cursor()
        cursor.execute("""
            UPDATE path_records 
            SET status = 'deleted', updated_at = CURRENT_TIMESTAMP
            WHERE id = ?
        """, (path_id,))
        self.connection.commit()
        
    def row_to_record(self, row):
        """将数据库行转换为记录对象"""
        return PathRecord(
            id=row['id'],
            path=row['path'],
            first_visit=datetime.fromisoformat(row['first_visit']),
            last_visit=datetime.fromisoformat(row['last_visit']),
            visit_count=row['visit_count'],
            total_time_spent=row['total_time_spent'],
            importance_score=row['importance_score'],
            is_favorite=bool(row['is_favorite']),
            tags=eval(row['tags']) if row['tags'] else [],
            source=row['source'],
            status=row['status']
        )
        
    def close(self):
        """关闭数据库连接"""
        if self.connection:
            self.connection.close()
```

#### 3.4.1 窗口监控器 (window_monitor.py)
```python
import win32gui
import win32con
import win32process
import psutil
from threading import Thread, Event
from queue import Queue

class WindowMonitor:
    """窗口监控器"""
    
    def __init__(self):
        self.is_running = False
        self.event_queue = Queue()
        self.stop_event = Event()
        self.monitor_thread = None
        self.hook = None
        
    def start(self):
        """启动监控"""
        if self.is_running:
            return
            
        self.is_running = True
        self.stop_event.clear()
        
        # 启动监控线程
        self.monitor_thread = Thread(target=self._monitor_loop, daemon=True)
        self.monitor_thread.start()
        
    def stop(self):
        """停止监控"""
        self.is_running = False
        self.stop_event.set()
        
        if self.hook:
            win32gui.UnhookWinEvent(self.hook)
            self.hook = None
            
        if self.monitor_thread:
            self.monitor_thread.join(timeout=1)
            
    def _monitor_loop(self):
        """监控循环"""
        # 设置窗口事件钩子
        self.hook = win32gui.SetWinEventHook(
            win32con.EVENT_SYSTEM_FOREGROUND,
            win32con.EVENT_SYSTEM_FOREGROUND,
            0,
            self._event_callback,
            0,
            0,
            win32con.WINEVENT_OUTOFCONTEXT
        )
        
        # 消息循环
        while not self.stop_event.is_set():
            win32gui.PumpWaitingMessages()
            self.stop_event.wait(0.1)
            
    def _event_callback(self, hWinEventHook, event, hwnd, idObject, idChild, dwEventThread, dwmsEventTime):
        """窗口事件回调"""
        try:
            # 检查是否是资源管理器窗口
            if self._is_explorer_window(hwnd):
                # 获取窗口信息
                window_info = {
                    'hwnd': hwnd,
                    'class_name': win32gui.GetClassName(hwnd),
                    'title': win32gui.GetWindowText(hwnd),
                    'process_id': win32process.GetWindowThreadProcessId(hwnd)[1]
                }
                
                # 放入事件队列
                self.event_queue.put(window_info)
        except Exception as e:
            # 忽略错误，继续监控
            pass
            
    def _is_explorer_window(self, hwnd):
        """判断是否是资源管理器窗口"""
        try:
            # 检查窗口类名
            class_name = win32gui.GetClassName(hwnd)
            if class_name == "CabinetWClass":
                return True
                
            # 检查进程名
            _, pid = win32process.GetWindowThreadProcessId(hwnd)
            process = psutil.Process(pid)
            return process.name().lower() == "explorer.exe"
        except:
            return False
            
    def get_next_event(self, timeout=None):
        """获取下一个事件"""
        try:
            return self.event_queue.get(timeout=timeout)
        except:
            return None
```

#### 3.4.2 路径过滤器 (path_filter.py)
```python
import re
import os
from typing import List, Pattern

class PathFilter:
    """路径过滤器"""
    
    def __init__(self, config=None):
        self.config = config or {}
        self.excluded_paths = self.config.get('excluded_paths', [])
        self.excluded_patterns = self.config.get('excluded_patterns', [])
        self.min停留时间 = self.config.get('min停留时间', 2000)
        self.max路径深度 = self.config.get('max路径深度', 10)
        
        # 编译正则表达式
        self.compiled_patterns = [re.compile(p, re.IGNORECASE) for p in self.excluded_patterns]
        
    def should_record(self, path) -> bool:
        """判断是否应该记录路径"""
        # 1. 检查路径深度
        if self._get_path_depth(path) > self.max路径深度:
            return False
            
        # 2. 检查排除路径
        if self._is_excluded_path(path):
            return False
            
        # 3. 检查排除模式
        if self._matches_excluded_pattern(path):
            return False
            
        # 4. 检查系统路径
        if self._is_system_path(path):
            return False
            
        return True
        
    def _get_path_depth(self, path) -> int:
        """获取路径深度"""
        return path.count('\\')
        
    def _is_excluded_path(self, path) -> bool:
        """检查是否是排除路径"""
        path_lower = path.lower()
        
        for excluded in self.excluded_paths:
            excluded_lower = excluded.lower()
            
            # 支持通配符 *
            if '*' in excluded_lower:
                # 转换为正则表达式
                pattern = excluded_lower.replace('*', '.*').replace('?', '.')
                if re.match(pattern, path_lower):
                    return True
            else:
                # 精确匹配或前缀匹配
                if path_lower == excluded_lower or path_lower.startswith(excluded_lower + '\\'):
                    return True
                    
        return False
        
    def _matches_excluded_pattern(self, path) -> bool:
        """检查是否匹配排除模式"""
        for pattern in self.compiled_patterns:
            if pattern.match(path):
                return True
        return False
        
    def _is_system_path(self, path) -> bool:
        """检查是否是系统路径"""
        system_paths = [
            'c:\\windows',
            'c:\\program files',
            'c:\\program files (x86)',
            'c:\\$recycle.bin',
            'c:\\system volume information',
            'c:\\recovery',
```

#### 3.4.3 去重处理器 (path_dedup.py)
```python
from collections import OrderedDict
from datetime import datetime, timedelta
from typing import Optional

class PathDeduplicator:
    """路径去重处理器"""
    
    def __init__(self, config=None):
        self.config = config or {}
        self.time_window = self.config.get('时间去重窗口', 5000)  # 毫秒
        self.max_recent_paths = self.config.get('最大最近路径', 20)
        
        # 最近访问的路径（有序字典）
        self.recent_paths = OrderedDict()
        
        # 最后访问时间
        self.last_access_times = {}
        
    def is_duplicate(self, path) -> bool:
        """判断是否是重复路径"""
        now = datetime.now()
        
        # 1. 时间去重
        if self._is_time_duplicate(path, now):
            return True
            
        # 2. 层级去重
        if self._is_hierarchical_duplicate(path):
            return True
            
        # 更新访问记录
        self._update_access_record(path, now)
        
        return False
        
    def _is_time_duplicate(self, path, now) -> bool:
        """基于时间的去重"""
        if path in self.last_access_times:
            last_access = self.last_access_times[path]
            time_diff = (now - last_access).total_seconds() * 1000
            
            # 在时间窗口内，视为重复
            if time_diff < self.time_window:
                return True
                
        return False
        
    def _is_hierarchical_duplicate(self, path) -> bool:
        """基于层级的去重"""
        if not self.recent_paths:
            return False
            
        last_path = next(reversed(self.recent_paths))
        
        # 检查是否是向上导航（从子目录返回父目录）
        if self._is_parent_path(path, last_path):
            # 如果父目录已经在最近列表中，视为重复
            if path in self.recent_paths:
                # 移动到最新位置
                self.recent_paths.move_to_end(path)
                return True
                
        # 检查是否是向下导航（从父目录进入子目录）
        elif self._is_parent_path(last_path, path):
            # 如果子目录已经在最近列表中，视为重复
            if path in self.recent_paths:
                self.recent_paths.move_to_end(path)
                return True
                
        return False
        
    def _is_parent_path(self, parent, child) -> bool:
        """判断是否是父路径"""
        parent_lower = parent.lower()
        child_lower = child.lower()
        return child_lower.startswith(parent_lower + '\\') or child_lower == parent_lower
        
    def _update_access_record(self, path, now):
        """更新访问记录"""
        # 更新最后访问时间
        self.last_access_times[path] = now
        
        # 更新最近路径列表
        if path in self.recent_paths:
            self.recent_paths.move_to_end(path)
        else:
            self.recent_paths[path] = now
            
        # 保持列表大小
        while len(self.recent_paths) > self.max_recent_paths:
            self.recent_paths.popitem(last=False)
```

```python
import math

class ImportanceCalculator:
    """重要性计算器"""
    
    def __init__(self, config=None):
        self.config = config or {}
        self.weights = self.config.get('weights', {
            'time': 0.4,
            'frequency': 0.3,
            'depth': 0.3
        })
        self.favorite_bonus = self.config.get('favorite_bonus', 100)
        
    def calculate(self, record) -> int:
        """计算重要性分数"""
        score = 100  # 基础分
        
        # 时间权重：停留时间越长，分数越高
        time_weight = min(record.total_time_spent / 60000.0, 1.0)  # 最大1分钟
        score += int(time_weight * 50 * self.weights['time'])
        
        # 频率权重：访问次数越多，分数越高
        freq_weight = min(record.visit_count / 10.0, 1.0)  # 最大10次
        score += int(freq_weight * 30 * self.weights['frequency'])
        
        # 深度权重：路径深度适中（3-5层）加分
        depth = record.path.count('\\')
        if 3 <= depth <= 5:
            score += int(20 * self.weights['depth'])
        elif depth > 5:
            score -= int(10 * self.weights['depth'])
            
        # 收藏加分
        if record.is_favorite:
            score += self.favorite_bonus
            
        return max(0, score)  # 确保分数不为负数
        
    def get_importance_level(self, score) -> str:
        """获取重要性级别"""
        if score >= 200:
            return "★★★"
        elif score >= 150:
            return "★★"
        elif score >= 100:
            return "★"
        else:
            return ""
```

```python
        self.hide_timer.stop()
        self.setWindowOpacity(1.0)

        """鼠标离开事件"""
        self.hide_timer.start(10000)  # 10秒后隐藏
        self.setWindowOpacity(0.7)
        
    def mousePressEvent(self, event: QMouseEvent):
        """鼠标按下事件"""
        if event.button() == Qt.LeftButton:
            self.dragging = True
            self.drag_position = event.globalPos() - self.frameGeometry().topLeft()
            event.accept()
            
    def mouseMoveEvent(self, event: QMouseEvent):
        """鼠标移动事件"""
        if self.dragging and event.buttons() & Qt.LeftButton:
            self.move(event.globalPos() - self.drag_position)
            event.accept()
            
    def mouseReleaseEvent(self, event: QMouseEvent):
        """鼠标释放事件"""
        if event.button() == Qt.LeftButton:
            self.dragging = False
            
            # 如果没有拖拽，则视为点击
            if not self.dragging:
                self.clicked_signal.emit()
                
    def mouseDoubleClickEvent(self, event: QMouseEvent):
        """鼠标双击事件"""
        if event.button() == Qt.LeftButton:
            self.clicked_signal.emit()
            
    def move_to_edge(self):
        """移动到屏幕边缘"""
        screen = QApplication.primaryScreen().geometry()
        x = screen.width() - 30
        y = screen.height() // 2
        self.move(x, y)
        
    def hide_to_tray(self):
        """隐藏到托盘"""
        self.hide()
        
    def showEvent(self, event):
        """显示事件"""
        super().showEvent(event)
        self.hide_timer.start(10000)
```
